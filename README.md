// Install Gulp with Browsersync

First install node https://nodejs.org/en/

// Install Gulp CLI
$ npm install gulp-cli -g

// Create package.json
$ npm init -y

// Install pakkages (node_modules)
$ npm install gulp browser-sync gulp-connect-php

// Start Gulp in project
$ gulp
